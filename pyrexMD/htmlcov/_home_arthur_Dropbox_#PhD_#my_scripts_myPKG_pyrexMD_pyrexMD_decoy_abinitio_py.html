28def frame2score(mobile, frames, save_as, sel="protein", warn=True): 

29 """ 

30 Get rosetta scores (ref2015) for frames of mobile.trajectory 

31 

32 Args: 

33 mobile 

34 frames (list): frame indices of mobile.trajectory 

35 save_as (str): logfile containing table of [FRAMES, SCORES] 

36 sel (str): selection string 

37 

38 Returns: 

39 FRAMES (list) 

40 list with frames of mobile.trajectory 

41 SCORES (list) 

42 list with rosetta scores (ref2015)

46 if isinstance(mobile, str):

57 mobile.trajectory[frame] 

58 structure = mobile.select_atoms(sel)

80def get_torsion_angles(obj, mol_type="auto", prec=2, spacing=12, verbose=True, verbose_stop=None): 

81 """ 

82 get torsion angles of object. 

83 

84 Args: 

85 obj (pose, str): pyrosetta pose object or pdb path 

86 mol_type (str): 

87 | molecule type 

88 | "auto": detect "protein" or "rna" by checking if pose.phi(1) exists. 

89 | "protein": protein molecule with torsion angles(phi, psi) 

90 | "rna": rna molecule with torsion angles(alpha, beta, gamma, delta) 

91 prec (int): rounding precision 

92 verbose (bool): print table with torsion angles 

93 

94 Returns: 

95 TORSION_ANGLES (tuple) 

96 | tuple containing lists with torsion angles of each residue 

97 | if mol_type == "protein": 

98 | TORSION_ANGLES = (phi, psi) 

99 | if mol_type == "rna": 

100 | TORSION_ANGLES = (alpha, beta, gamma, delta) 

101 """ 

102 if isinstance(obj, str): 

103 pose = pyrosetta.pose_from_pdb(obj) 

104 else: 

105 pose = obj 

106 

107 if mol_type.lower() == "auto": 

108 try: 

109 pose.phi(1) 

110 mol_type = "protein" 

111 except RuntimeError: 

112 mol_type = "rna" 

113 

114 if mol_type.lower() == "protein": 

115 PHI, PSI, = [], [] 

116 

117 for i in range(1, 1+pose.total_residue()): 

118 PHI.append(round(pose.phi(i), prec)) 

119 PSI.append(round(pose.psi(i), prec)) 

120 

121 TORSION_ANGLES = (PHI, PSI) 

122 if verbose: 

123 _misc.cprint("resid \tphi \tpsi".expandtabs(spacing), "blue") 

124 RESID = list(range(1, 1+pose.total_residue())) 

125 _misc.print_table([RESID, PHI, PSI], 

126 prec=prec, spacing=spacing, verbose_stop=verbose_stop) 

127 

128 if mol_type.lower() == "rna": 

129 ALPHA, BETA, GAMMA, DELTA = [], [], [], [] 

130 

131 for i in range(1, 1+pose.total_residue()): 

132 ALPHA.append(round(pose.alpha(i), prec)) 

133 BETA.append(round(pose.beta(i), prec)) 

134 GAMMA.append(round(pose.gamma(i), prec)) 

135 DELTA.append(round(pose.delta(i), prec)) 

136 

137 TORSION_ANGLES = (ALPHA, BETA, GAMMA, DELTA) 

138 if verbose: 

139 _misc.cprint("resid \talpha \tbeta \tgamma \tdelta".expandtabs(spacing), "blue") 

140 RESID = list(range(1, 1+pose.total_residue())) 

141 _misc.print_table([RESID, ALPHA, BETA, GAMMA, DELTA], 

142 prec=prec, spacing=spacing, verbose_stop=verbose_stop) 

143 

144 return TORSION_ANGLES

147def apply_relative_torsion_angle(pose_ref, pose, resid=1, angle="alpha", shift=0, verbose=True): 

148 """ 

149 Apply relative torsion angle(use pose_ref angle and apply relative shift). 

150 

151 Args: 

152 pose_ref (pose) 

153 pose (pose) 

154 resid (int) 

155 angle (str) 

156 | protein: "phi", "psi" 

157 | rna: "alpha", "beta", "gamma", "delta" 

158 shift (float): relative shift value 

159 verbose (bool): print applied changes to torsion angle. 

160 """ 

161 with _misc.HiddenPrints(verbose=verbose): 

162 if angle.lower() == "phi": 

163 set_angle = pose.set_phi 

164 vref = pose_ref.phi(resid) 

165 v1 = pose.phi(resid) 

166 elif angle.lower() == "psi": 

167 set_angle = pose.set_psi 

168 vref = pose_ref.psi(resid) 

169 v1 = pose.psi(resid) 

170 elif angle.lower() == "alpha": 

171 set_angle = pose.set_alpha 

172 vref = pose_ref.alpha(resid) 

173 v1 = pose.alpha(resid) 

174 elif angle.lower() == "beta": 

175 set_angle = pose.set_beta 

176 vref = pose_ref.beta(resid) 

177 v1 = pose.beta(resid) 

178 elif angle.lower() == "gamma": 

179 set_angle = pose.set_gamma 

180 vref = pose_ref.gamma(resid) 

181 v1 = pose.gamma(resid) 

182 elif angle.lower() == "delta": 

183 set_angle = pose.set_delta 

184 vref = pose_ref.delta(resid) 

185 v1 = pose.delta(resid) 

186 

187 v2 = vref + shift 

188 set_angle(resid, v2) 

189 print(f"resid {resid} {angle:>5}: {v1:>8.2f} --> {v2:>8.2f} (ref:{vref:>8.2f})")

290 """ 

291 Create decoys within pyrosetta framework. 

292 

293 Args: 

294 abinitio_cfg (CONFIG class): output of abinitio.setup_abinitio_cfg() 

295 output_dir (str): output directory for decoys 

296 stream2pymol (bool): stream decoys to pymol 

297 fastrelax (bool): apply fastrelax protocol on decoys before dumping them as pdb 

298 save_log (bool): save scores to logfile at < output_dir/scores.txt >

301 cprint_color (None, str): colored print color

423def _HELP_decoypath_to_decoyid(list_): 

424 """ 

425 convert list with decoy strings to list with decoy ids 

426 

427 Example: 

428 | >> decoy_list = ['./1LMB_decoys/set1_2000/1LMB_1.pdb', 

429 | './1LMB_decoys/set1_2000/1LMB_2.pdb',

431 | >> _HELP_decoystr_to_decoyid(decoy_list) 

432 | [1, 2, 3] 

433 """ 

434 substrings = [_misc.get_substrings(i) for i in list_] 

435 

436 # get decoy index range 

437 diff_element = list(set(substrings[0])-set(substrings[1])) 

438 if len(diff_element) != 1: 

439 raise ValueError('substrings differ in more than 1 element (i.e. differ at more than 1 position)') 

440 diff_position = substrings[0].index(diff_element[0]) 

441 ids = [int(_misc.get_substrings(i)[diff_position]) for i in list_] 

442 return ids

445def get_decoy_list(decoy_dir, pattern="*.pdb", ndx_range=(None, None)): 

446 """ 

447 | Alias function of get_structure_list(). 

448 | get decoy list (sorted by a numeric part at any position of the filename, 

449 | e.g. 1LMB_1.pdb, 1LMB_2.pdb, ...) 

450 

451 Args: 

452 decoy_dir (str): decoy directory 

453 pattern (str): pattern of decoy filenames 

454 ndx_range (tuple, list): limit decoy index range to [ndx_min, ndx_max] 

455 

456 Returns: 

457 DECOY_LIST (list) 

458 list with decoy filenames 

459 """ 

460 if not isinstance(decoy_dir, str): 

461 raise TypeError("type(decoy_dir) must be string.") 

462 if decoy_dir[-1] == "/": 

463 decoy_dir = decoy_dir[:-1] 

464 

465 # preset stuff 

466 decoy_list = glob.glob(f"{decoy_dir}/{pattern}") 

467 decoy_names = [_misc.get_filename(i) for i in decoy_list] 

468 decoy_names_substrings = [_misc.get_substrings(i) for i in decoy_names] 

469 

470 # get decoy index range 

471 diff_element = list(set(decoy_names_substrings[0])-set(decoy_names_substrings[1])) 

472 if len(diff_element) != 1: 

473 raise ValueError('decoy_names_substrings differ in more than 1 element (i.e. differ at more than 1 position)') 

474 diff_position = decoy_names_substrings[0].index(diff_element[0]) 

475 min_ndx = 999999999 

476 max_ndx = -999999999 

477 for i in decoy_names_substrings: 

478 if i[diff_position].isdigit() and int(i[diff_position]) < min_ndx: 

479 min_ndx = int(i[diff_position]) 

480 if i[diff_position].isdigit() and int(i[diff_position]) > max_ndx: 

481 max_ndx = int(i[diff_position]) 

482 

483 if ndx_range[0] is not None: 

484 min_ndx = ndx_range[0] 

485 if ndx_range[1] is not None: 

486 max_ndx = ndx_range[1] 

487 

488 # get decoy filename prefix and suffix 

489 template = decoy_names[0] 

490 template_diff_element = _misc.get_substrings(template)[diff_position] 

491 decoy_filename_prefix = template[:template.index(template_diff_element)] 

492 decoy_filename_suffix = template[template.index(template_diff_element)+len(template_diff_element):] 

493 

494 # create sorted decoy list 

495 DECOY_LIST = [f"{decoy_dir}/{decoy_filename_prefix}{i}{decoy_filename_suffix}" for i in range(min_ndx, max_ndx+1) 

496 if f"{decoy_dir}/{decoy_filename_prefix}{i}{decoy_filename_suffix}" in decoy_list] 

497 return DECOY_LIST

562 else: 

563 raise TypeError("Specify either decoy_list or decoy_dir.") 

564 DECOY_ID = _HELP_decoypath_to_decoyid(DECOY_LIST) 

565 SCORE = [] # list with corresponding scores 

566 

567 if verbose: 

568 _misc.cprint("Getting scores...", cfg.cprint_color) 

569 for decoy_path in tqdm(DECOY_LIST, disable=not verbose): 

570 if not os.path.exists(decoy_path): 

571 print(f"The decoy path '{decoy_path}' does not exist!") 

572 continue 

573 

574 # get SCORE 

575 pose = pyrosetta.pose_from_pdb(decoy_path) 

576 SCORE.append(scorefxn(pose)) 

577 

578 if n_rank is not None: 

579 DECOY_LIST = [DECOY_LIST[i] for i in _misc.get_ranked_array(SCORE, reverse=reverse, verbose=False)[1][:n_rank]] 

580 DECOY_ID = [DECOY_ID[i] for i in _misc.get_ranked_array(SCORE, reverse=reverse, verbose=False)[1][:n_rank]] 

581 SCORE = [SCORE[i] for i in _misc.get_ranked_array(SCORE, reverse=reverse, verbose=False)[1][:n_rank]] 

582 return DECOY_LIST, DECOY_ID, SCORE

585def get_decoy_RMSD(ref, decoy_list=None, decoy_dir=None, pattern="*.pdb", 

586 ndx_range=(None, None), n_rank=None, reverse=True, 

587 sel='backbone', verbose=True, **kwargs): 

588 """ 

589 Get decoy RMSD of < decoy_list > or < decoy_dir > . 

590 

591 - If < n_rank > is None: return unranked decoys, ids and RMSD. 

592 - If < n_rank > is int: return ranked decoys, ids and RMSD for best < n_rank > decoys. 

593 

594 Args: 

595 ref (universe): reference structure 

596 decoy_list (None, list): list with decoy paths 

597 decoy_dir (None, str): decoy directory 

598 pattern (str): pattern of decoy filenames 

599 ndx_range (tuple, list): limit decoy index range to[ndx_min, ndx_max] 

600 n_rank (None, int): 

601 | None: return unranked decoys, ids and scores. 

602 | int: return ranked decoys, ids and scores for best < n_rank > decoys. 

603 reverse (bool): 

604 | True: ascending ranking order(low to high) 

605 | False: decending ranking order(high to low) 

606 verbose (bool) 

607 

608 Keyword Args: 

609 cprint_color (None, str): colored print color 

610 

611 Returns: 

612 DECOY_LIST (list) 

613 list with decoy names 

614 DECOY_ID (list) 

615 list with decoy ids 

616 RMSD (list) 

617 list with decoy RMSDs 

618 """ 

619 default = {"cprint_color": "blue"} 

620 cfg = _misc.CONFIG(default, **kwargs) 

621 ############################################################################ 

622 if decoy_list is None and decoy_dir is None: 

623 raise TypeError("Specify either decoy_list or decoy_dir.") 

624 elif isinstance(decoy_list, list): 

625 DECOY_LIST = decoy_list 

626 elif isinstance(decoy_dir, str): 

627 DECOY_LIST = get_decoy_list(decoy_dir, pattern, ndx_range) 

628 else: 

629 raise TypeError("Specify either decoy_list or decoy_dir.") 

630 DECOY_ID = _HELP_decoypath_to_decoyid(DECOY_LIST) 

631 RMSD = [] # list with corresponding RMSD values 

632 

633 if verbose: 

634 _misc.cprint("Getting RMSDs...", cfg.cprint_color) 

635 for decoy_path in tqdm(DECOY_LIST, disable=not verbose): 

636 if not os.path.exists(decoy_path): 

637 print(f"The decoy path '{decoy_path}' does not exist!") 

638 continue 

639 

640 # get RMSD 

641 decoy = mda.Universe(decoy_path, dt=0) # dt=0 fixes warnings.warn("Reader has no dt information, set to 1.0 ps") of MDAnalysis/coordinates/base.py 

642 sel1, sel2 = _ana.get_matching_selection(decoy, ref, sel=sel, norm=True, verbose=False) 

643 ftr = _ana.get_RMSD(decoy, ref, sel1=sel1, sel2=sel2) 

644 RMSD.append(ftr[2]) 

645 

646 RMSD = _misc.flatten_array(RMSD) 

647 

648 if n_rank is not None: 

649 DECOY_LIST = [DECOY_LIST[i] for i in _misc.get_ranked_array(RMSD, reverse=reverse, verbose=False)[1][:n_rank]] 

650 DECOY_ID = [DECOY_ID[i] for i in _misc.get_ranked_array(RMSD, reverse=reverse, verbose=False)[1][:n_rank]] 

651 RMSD = [RMSD[i] for i in _misc.get_ranked_array(RMSD, reverse=reverse, verbose=False)[1][:n_rank]] 

652 return DECOY_LIST, DECOY_ID, RMSD

683def rank_decoy_precision(data, rank_by="SCORE", verbose=True): 

684 """ 

685 rank decoy precision 

686 

687 Args: 

688 data (list): 

689 | output of abinitio.get_decoy_precision() 

690 | data[0]: DECOY_LIST 

691 | data[1]: DECOY_ID 

692 | data[2]: SCORE 

693 | data[3]: RMSD 

694 rank_by (str): "SCORE", "RMSD" 

695 verbose (bool) 

696 

697 Returns: 

698 DECOY_LIST_ranked (array) 

699 ranked list with decoy paths 

700 DECOY_ID_ranked (array) 

701 ranked list with decoy ids 

702 SCORE_ranked (array) 

703 ranked list with corresponding scores (ref 2015) 

704 RMSD_ranked (array) 

705 ranked list with corresponding RMSD values 

706 """ 

707 DECOY_LIST, DECOY_ID, SCORE, RMSD = data 

708 if rank_by.upper() == "SCORE": 

709 SCORE_ranked, SCORE_ranked_ndx = _misc.get_ranked_array(SCORE, reverse=True, verbose=verbose) 

710 DECOY_LIST_ranked = np.array([DECOY_LIST[ndx] for ndx in SCORE_ranked_ndx]) 

711 DECOY_ID_ranked = np.array([DECOY_ID[ndx] for ndx in SCORE_ranked_ndx]) 

712 RMSD_ranked = np.array([RMSD[ndx] for ndx in SCORE_ranked_ndx])

717 SCORE_ranked = np.array([SCORE[ndx] for ndx in RMSD_ranked_ndx]) 

718 

719 return DECOY_LIST_ranked, DECOY_ID_ranked, SCORE_ranked, RMSD_ranked
