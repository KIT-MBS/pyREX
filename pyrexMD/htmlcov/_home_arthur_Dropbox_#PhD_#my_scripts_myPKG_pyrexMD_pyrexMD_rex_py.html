62 create_dir=True, verbose=False, **kwargs): 

63 """ 

64 Apply forcefield on best decoys and save as <filename>_ref.pdb.

261def WF_getParameter_boxsize(logfile="./logs/editconf.log", base=0.2, verbose=True): 

262 """ 

263 Read <logfile> and suggest a 'fixed boxsize' parameter for REX simulations. 

264 

265 Args: 

266 logfile (str): path to <editconf.log> containing the line 'system size: X Y Z' 

267 base (float): round up base for highest box dimension taken from <logfile> 

268 verbose (bool)prep_REX_tpr 

269 

270 Returns: 

271 boxsize (float) 

272 suggested boxsize parameter 

273 """ 

274 boxsize = None 

275 

276 with _misc.HiddenPrints(verbose=verbose): 

277 with open(logfile, "r") as fin: 

278 _misc.cprint(f"Reading logfile: {logfile}") 

279 for line in fin: 

280 

281 if "new box vectors" in line: 

282 s = line.split() 

283 dims = [float(x) for x in s if x.replace(".", "", 1).isdigit()] 

284 boxsize = round(_misc.round_up(max(dims), base=base), 2) 

285 _misc.cprint(line, "blue", end='') 

286 

287 elif "system size" in line or "diameter" in line or "box volume" in line: 

288 _misc.cprint(line, "blue", end='') 

289 

290 if boxsize is None: 

291 raise _misc.Error("No boxsize parameters found.") 

292 else: 

293 _misc.cprint(f"suggested box size: {boxsize}", "green", end='') 

294 return boxsize

297def WF_getParameter_maxsol(logfile="./logs/solvate.log", maxsol_reduce=50, verbose=True): 

298 """ 

299 Read <logfile> and suggest a 'max solution' parameter for REX simulations. 

300 

301 Args: 

302 logfile (str): path to <editconf.log> containing the line 'system size: X Y Z' 

303 maxsol_reduce (int): 

304 | reduce max solution number taken from <logfile> 

305 | -> guarantees fixed solution number for different start configurations 

306 verbose (bool) 

307 

308 Returns: 

309 maxsol (int) 

310 suggested max solution parameter 

311 """ 

312 maxsol = None 

313 with open(logfile, "r") as fin: 

314 for line in fin: 

315 if "Number of solvent molecules" in line: 

316 s = line.split() 

317 maxsol = [int(x)-maxsol_reduce for x in s if x.isdigit()][0] 

318 

319 if verbose: 

320 _misc.cprint(f"Reading logfile: {logfile}") 

321 _misc.cprint(line, "blue", end="") 

322 _misc.cprint(f"suggested max solution: {maxsol}", "green", end="") 

323 

324 if maxsol is None: 

325 raise _misc.Error("No solution parameter found.") 

326 return maxsol

329def WF_REX_setup(rex_dirs, boxsize, maxsol, verbose=False, verbose_gmx=False): 

330 """ 

331 Workflow: REX setup (without energy minimization) 

332 

333 Args: 

334 rex_dirs (list): list with rex_dirs (output of rex.get_REX_DIRS()) 

335 boxsize (float): suggested boxsize parameter (output of gmx.WF_getParameter_boxsize()) 

336 maxsol (int): suggested max solution parameter (output of gmx.WF_getParameter_max_solution()) 

337 verbose (bool): show blue log messages (saved file as, saved log as) 

338 verbose_gmx (bool): show gmx module messages (requires verbose=True) 

339 """ 

340 for rex_dir in rex_dirs: 

341 _misc.cprint("#######################################################################################") 

342 _misc.cd(rex_dir) 

343 decoy_pdb = _misc.get_filename("*_ref.pdb") 

344 _misc.cprint(f"Using decoy pdb: {decoy_pdb}") 

345 

346 # 1) generate topology 

347 _misc.cprint(f"\nGenerating topology...", "red") 

348 with _misc.HiddenPrints(verbose=verbose): 

349 protein_gro = _gmx.pdb2gmx(f=decoy_pdb, verbose=False) 

350 

351 # 2) generate box 

352 _misc.cprint(f"Generating box with fixed size ({boxsize}) ...", "red") 

353 with _misc.HiddenPrints(verbose=verbose): 

354 _gmx.editconf(f=protein_gro, o="box.gro", bt="cubic", box=boxsize, c=True, verbose=verbose_gmx) 

355 

356 # 3) generate solvent 

357 _misc.cprint(f"Generating solvent with fixed solvent molecules ({maxsol})...", "red") 

358 with _misc.HiddenPrints(verbose=verbose): 

359 _gmx.solvate(cp="box.gro", maxsol=maxsol, verbose=verbose_gmx) 

360 

361 # 4) generate ions 

362 _misc.cprint(f"Generating ions...", "red") 

363 with _misc.HiddenPrints(verbose=verbose): 

364 _gmx.grompp(f="../ions.mdp", o="ions.tpr", c="solvent.gro", p="topol.top", verbose=verbose_gmx) 

365 _gmx.genion(s="ions.tpr", o="ions.gro", p="topol.top", verbose=False) 

366 

367 _misc.cprint("#######################################################################################") 

368 _misc.cd("..") 

369 _misc.cprint("Finished setup of all REX DIRS (skipped energy minimization).", "green") 

370 return

449def prep_REX_temps(T_0=None, n_REX=None, k=None): 

450 """ 

451 Prepare REX temperatures. 

452 

453 Args: 

454 T_0 (None/float): starting temperature (Kelvin) 

455 n_REX (None/int): number of replica 

456 k (None/float): temperature distrubtion scaling factor 

457 """ 

458 T1 = [] # list with temperatures for method 1 

459 T2 = [] # list with temperatures for method 2 

460 

461 if T_0 is None: 

462 T_0 = int(input("Enter starting temperature T_0 (Kelvin): ")) 

463 else: 

464 print("Enter starting temperature T_0 (Kelvin): ", T_0) 

465 if n_REX is None: 

466 n_REX = int(input("Enter replica number n_REX: ")) 

467 else: 

468 print("Enter replica number n_REX: ", n_REX) 

469 if k is None: 

470 k = float(input("Enter scaling factor k: ")) 

471 else: 

472 print("Enter scaling factor k: ", k) 

473 

474 # method 1: same function for all replica 

475 print("\nMethod 1:") 

476 print("REX Temperature Distribution: T_i = T_0*exp(k*i)") 

477 

478 with open("rex_temps.log", "w") as fout: 

479 _misc.cprint(f"Saved log as: {_misc.realpath('rex_temps.log')}", "blue") 

480 fout.write("Function: T_i = {}*exp({}*i)\n".format(T_0, k)) 

481 fout.write("Temperatures:\n") 

482 for i in range(0, n_REX): 

483 T_i = format(1.00*T_0*np.exp(k*i), ".2f") 

484 T1.append(T_i) 

485 fout.write("{}, ".format(T_i)) 

486 

487 # method 2: same as method 1 but with adjusting the coefficients ai 

488 # (ai increase spacing of higher temperatures to get lower exchange probabilities) 

489 

490 a = [] # lin. stretching factors 

491 delta_a = 0.04 

492 

493 # increase a by delta_a every 10 replica 

494 if n_REX % 10 == 0: 

495 for i in range(n_REX//10): 

496 if i == 0: 

497 a.append(1.00) 

498 else: 

499 a.append(round(a[-1]+delta_a, 2)) 

500 else: 

501 for i in range((n_REX//10)+1): 

502 if i == 0: 

503 a.append(1.00) 

504 else: 

505 a.append(round(a[-1]+delta_a, 2)) 

506 

507 print("\nMethod 2") 

508 print("REX Temperature Distribution:") 

509 print("T_0 = {} K ; DELTA = T_0 * (exp(k*i)-exp(k*(i-1)))".format(T_0)) 

510 print("T_i = T_(i-1) + a_i * DELTA") 

511 

512 with open("rex_temps.log", "w") as fout: # same fout name as method 1 -> will overwrite 

513 _misc.cprint(f"Saved log as: {_misc.realpath('rex_temps.log')}", "blue") 

514 fout.write("REX Temperature Distribution:\n") 

515 fout.write("T_0 = {} K ; DELTA = T_0 * (exp(k*i)-exp(k*(i-1)))\n".format(T_0)) 

516 fout.write("T_i = T_(i-1) + a_j * DELTA\n") 

517 fout.write("\nChosen Parameter:\n") 

518 fout.write("k = {}\n".format(k)) 

519 for i in range(len(a)): 

520 fout.write("a_{} = {:.2f} for i = {}..{} \n".format(i, a[i], i*10, i*10+9)) 

521 

522 fout.write("\nTemperatures:\n") 

523 for i in range(0, n_REX): 

524 if i == 0: 

525 T2.append(T_0) # start value 300 K 

526 else: 

527 index = i//10 # find out which a[index] to use 

528 delta = T_0*(np.exp(k*i) - np.exp(k*(i-1))) 

529 T2.append(T2[i-1] + a[index]*delta) 

530 

531 T2 = [format(x, ".2f") for x in T2] 

532 for i in range(len(T2)): 

533 fout.write("{}, ".format(T2[i])) 

534 

535 # comparison of method 1 and 2 

536 print("\nTemperatures Method 1:") 

537 print(T1) 

538 print("\nTemperatures Method 2:") 

539 print(T2) 

540 

541 # DELTA TEST 

542 DELTA1 = [] 

543 DELTA2 = [] 

544 DELTA_DELTA = [] 

545 

546 for i in range(len(T1)-1): 

547 DELTA1.append(format(float(T1[i+1])-float(T1[i]), ".2f")) 

548 print("\nDelta Temps Method 1 (DTM1):") 

549 print(DELTA1) 

550 

551 for i in range(len(T2)-1): 

552 DELTA2.append(format(float(T2[i+1])-float(T2[i]), ".2f")) 

553 print("\nDelta Temps Method 2 (DTM2):") 

554 print(DELTA2) 

555 

556 for i in range(len(DELTA1)): 

557 DELTA_DELTA.append(format(float(DELTA2[i])-float(DELTA1[i]), ".2f"))

562def prep_REX_mdp(main_dir="./", template="rex.mdp", n_REX=None, verbose=True): 

563 """ 

564 Prepare REX mdp -> copy template and change tempereratures according to rex_temps.log 

565 

566 Args: 

567 main_dir (str): main directory with rex_1, rex_2, etc. 

568 template (str): template file 

569 n_REX (None/int): number of replica 

570 verbose (bool) 

571 """ 

572 with open("rex_temps.log", "r") as fin: 

573 for line in fin: 

574 # search for REX temperatures (should be last entry) 

575 s = line.split(",") 

576 REX_TEMPS = [x.lstrip() for x in s if "." in x and x.lstrip().replace(".", "", 1).isdigit()] 

577 

578 # create N rex.mdp files and edit the ref_t line 

579 if n_REX is None: 

580 n_REX = int(input("Enter n_REX:")) 

581 else: 

582 print("Enter n_REX:", n_REX) 

583 print(f"Using {template} as template and changing temperatures according to rex_temps.log...") 

584 

585 for i in range(1, n_REX+1): 

586 file_path = _misc.mkdir("rex_" + str(i)) 

587 with open(_misc.joinpath(main_dir, template), "r") as fin, open(file_path + f"/{template}", "w") as fout: 

588 if verbose: 

589 print("Saved mdp file as: " + file_path + f"/{template}") 

590 

591 for line in fin: 

592 if "ref_t" in line: 

593 fout.write("ref_t = %s %s ; reference temperature, one for each group, in K\n" % (REX_TEMPS[i-1], REX_TEMPS[i-1])) 

594 elif "gen_temp" in line: 

595 fout.write("gen_temp = %s ; temperature for Maxwell distribution\n" % (REX_TEMPS[i-1])) 

596 else: 

597 fout.write(line) 

598 

599 return
